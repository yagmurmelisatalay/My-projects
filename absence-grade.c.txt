#include <stdio.h>

int main() {
    int grade;
    int absence;
    printf("What's your grade(100/?)\n");
    scanf("%d" , &grade);
    
    printf("How many absences do you have?\n");
    scanf("%d" , &absence); 
    
    if (grade<50 || absence>=10) {
        printf("You failed.\n");
    } else {
        printf("You passed.\n");
}

    return 0;

}
